1.2 Technology Stack Breakdown
Frontend (Mobile App)
	•	Framework: React Native 0.73
	•	Build Tool: Expo SDK 50
	•	State Management: React Context + AsyncStorage
	•	Navigation: React Navigation 6.x
	•	Audio: expo-av
	•	File System: expo-file-system
	•	Payments: react-native-razorpay
	•	UI Components: React Native Paper
Backend (API Server)
	•	Runtime: Node.js 20 LTS
	•	Framework: Express 4.x
	•	Database: MongoDB 7.0 + Mongoose 8.x
	•	Authentication: JWT + Firebase Auth
	•	File Upload: Multer
	•	PDF Generation: PDFKit
	•	Payment: Razorpay Node SDK
Third-Party Services
	•	Speech-to-Text: OpenAI Whisper API
	•	AI Processing: Anthropic Claude API
	•	Storage: Cloudflare R2 (or AWS S3)
	•	Phone Auth: Firebase Auth
	•	Payments: Razorpay

const UserSchema = new mongoose.Schema({   // Identity   _id: mongoose.Schema.Types.ObjectId,   phone: {     type: String,     required: true,     unique: true,     match: /^[6-9]\d{9}$/  // Indian phone number   },   countryCode: {     type: String,     default: '+91'   },      // Account Type   accountType: {     type: String,     enum: ['personal', 'verified_shop', 'premium'],     default: 'personal'   },      // Personal Info   name: {     type: String,     required: true,     trim: true   },   email: {     type: String,     lowercase: true,     trim: true   },      // Shop Details (for verified/premium accounts)   shopName: String,   shopAddress: String,   gstNumber: {     type: String,     match: /^[0-9]{2}[A-Z]{5}[0-9]{4}[A-Z]{1}[1-9A-Z]{1}Z[0-9A-Z]{1}$/   },      // Verification   isVerified: {     type: Boolean,     default: false   },   verificationDocuments: [{     type: String,  // Cloudflare R2 URL     uploadedAt: Date   }],   verifiedAt: Date,   verifiedBy: mongoose.Schema.Types.ObjectId,  // Admin who verified      // Subscription   subscriptionTier: {     type: String,     enum: ['free', 'verified', 'premium'],     default: 'free'   },   subscriptionStart: Date,   subscriptionExpiry: Date,   subscriptionId: String,  // Razorpay subscription ID      // Usage Limits (for free tier)   monthlyInvoiceCount: {     type: Number,     default: 0   },   lastResetDate: {     type: Date,     default: Date.now   },      // Security   firebaseUid: String,   lastLogin: Date,   deviceTokens: [String],  // For push notifications      // Metadata   createdAt: {     type: Date,     default: Date.now   },   updatedAt: {     type: Date,     default: Date.now   },   isActive: {     type: Boolean,     default: true   } });  // Indexes UserSchema.index({ phone: 1 }); UserSchema.index({ email: 1 }); UserSchema.index({ accountType: 1 }); UserSchema.index({ subscriptionExpiry: 1 });  module.exports = mongoose.model('User', UserSchema);
